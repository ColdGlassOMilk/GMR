Quick Start

# main.rb - A simple GMR game
include GMR

def init
  @sprite = Sprite.new("assets/player.png")
  @sprite.x = 160
  @sprite.y = 120
end

def update
  if Input.key_down?(:right)
    @sprite.x += 2
  end
end

def draw
  Graphics.clear([20, 20, 40])
  @sprite.draw
end
